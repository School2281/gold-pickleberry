server {
    listen 80;
    server_name raspberrypi.local;
    root /var/www/video_player;
    index index.html;
    
    # VULNERABLE BUFFER SETTINGS
    client_body_buffer_size 1k;      # Tiny buffer (normally 8k-16k)
    client_header_buffer_size 1k;    # Tiny header buffer
    large_client_header_buffers 2 1k;# Minimal buffers for large headers
    
    # SLOWLORIS-ENABLING SETTINGS
    client_body_timeout 3600s;      # 1 hour timeout for body
    client_header_timeout 3600s;    # 1 hour timeout for headers
    send_timeout 3600s;             # 1 hour timeout between packets
    
    # DISABLE RATE LIMITING
    # Don't add any 'limit_req' or 'limit_conn' directives
    
    location / {
        try_files $uri $uri/ =404;
    }
    
    location ~ \.mp4$ {
        mp4;
        mp4_buffer_size 500M;       # Huge buffer to waste memory
        mp4_max_buffer_size 1G;     # Even larger max buffer
        expires 1h;
    }
}
